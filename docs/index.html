Installation

Download: https://unpkg.com/hyper-auto-layout/dist/hyper-auto-layout.min.css
<!-- CDN Development (always latest) -->
<link rel="stylesheet" href="https://unpkg.com/hyper-auto-layout/dist/hyper-auto-layout.min.css"/>

<!-- CDN Production (specific release) -->
<link rel="stylesheet" href="https://unpkg.com/hyper-auto-layout@1.0.1/dist/hyper-auto-layout.min.css"/>

# npm package
$ npm install --save-dev hyper-auto-layout

/* CSS import */
@import url("https://unpkg.com/hyper-auto-layout@1.0.1/dist/hyper-auto-layout.min.css");
